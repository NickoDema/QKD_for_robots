  geometry_msgs

)

)

  Set_key.srv
  Aes.srv
)

)

  DEPENDS libcrypto++9
)

###########

###########

include_directories(
  include
  src
  ${catkin_INCLUDE_DIRS}
)

  src/kvant_slave_node.cpp
  src/kvant_slave.cpp
  src/kvant_basic.cpp
)

target_link_libraries(slave
  ${catkin_LIBRARIES}  /usr/lib/libcrypto++.a
)

add_executable( master
  src/kvant_master_node.cpp
  src/kvant_master.cpp
  src/kvant_basic.cpp
)

  ${catkin_LIBRARIES} /usr/lib/libcrypto++.a
